const express = require('express');
const cors = require('cors');
const pessoaRoutes = require('./routes/pessoaRoutes');

const app = express();
const PORT = 3000;

app.use(cors());
app.use(express.json());

app.use('/api', pessoaRoutes);

app.listen(PORT, () => {
  console.log(`Servidor rodando na porta ${PORT}`);
});
